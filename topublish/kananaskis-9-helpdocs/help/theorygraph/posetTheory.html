     poset (s,r) ⇔

     up_continuous (s,r) f ⇔
     ∀c x.
       chain (s,r) c ∧ lub (s,r) c x ⇒
       lub (s,r) (λy. ∃z. (s z ∧ c z) ∧ (y = f z)) (f x)
