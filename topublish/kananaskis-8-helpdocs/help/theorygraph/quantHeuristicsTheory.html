   ((X = (x,y)) ⇔ (FST X = x) ∧ (SND X = y))
PAIR_EQ_SIMPLE_EXPAND 
|- (((x,y) = (x,y')) ⇔ (y = y')) ∧ (((y,x) = (y',x)) ⇔ (y = y')) ∧
   (((FST X,y) = X) ⇔ (y = SND X)) ∧ (((x,SND X) = X) ⇔ (x = FST X)) ∧
   ((X = (FST X,y)) ⇔ (SND X = y)) ∧ ((X = (x,SND X)) ⇔ (FST X = x))
IS_SOME_EQ_NOT_NONE 
|- ∀x. IS_SOME x ⇔ x ≠ NONE
