iSQRT0_def 
⊢ ∀n.
      iSQRT0 n =
      (let
         p = SQRTd n ;
         d = SND p − FST p
       in
         if d = 0 then (2 * FST p,4 * SND p) else (SUC (2 * FST p),4 * d − 1))
iSQRT1_def 
⊢ ∀n.
      iSQRT1 n =
      (let
         p = SQRTd n ;
         d = SUC (SND p) − FST p
       in
         if d = 0 then (2 * FST p,SUC (4 * SND p))
         else (SUC (2 * FST p),4 * (d − 1)))
iSQRT2_def 
⊢ ∀n.
      iSQRT2 n =
      (let
         p = SQRTd n ;
         d = 2 * FST p ;
         c = SUC (2 * SND p) ;
         e = c − d
       in
         if e = 0 then (d,2 * c) else (SUC d,2 * e − 1))
iSQRT3_def 
⊢ ∀n.
      iSQRT3 n =
      (let
         p = SQRTd n ;
         d = 2 * FST p ;
         c = SUC (2 * SND p) ;
         e = SUC c − d
       in
         if e = 0 then (d,SUC (2 * c)) else (SUC d,2 * (e − 1)))
