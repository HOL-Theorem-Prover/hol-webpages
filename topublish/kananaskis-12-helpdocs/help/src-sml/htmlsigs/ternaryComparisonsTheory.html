   [char_compare_def]  Definition
      
      ⊢ ∀c1 c2. char_compare c1 c2 = num_compare (ORD c1) (ORD c2)

        (invert_comparison LESS = GREATER) ∧
        (invert_comparison EQUAL = EQUAL)
   
   [num_compare_def]  Definition
      
      ⊢ ∀n1 n2.
            num_compare n1 n2 =
