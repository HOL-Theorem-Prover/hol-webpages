      ⊢ ∀s. ARBITRARY s ⇔ T
   
   [INTERSECTION_OF]  Definition
      
      ⊢ ∀P Q.
          P INTERSECTION_OF Q =
          (λs. ∃u. P u ∧ (∀c. c ∈ u ⇒ Q c) ∧ BIGINTER u = s)
   
   [UNION_OF]  Definition
      
      ⊢ ∀P Q.
          P UNION_OF Q = (λs. ∃u. P u ∧ (∀c. c ∈ u ⇒ Q c) ∧ BIGUNION u = s)
   
   [closed_DEF]  Definition
      
      ⊢ ∀s. closed s ⇔ closed_in s 𝕌(:α)
   
   [closed_in]  Definition
      
      ⊢ ∀top s.
          closed_in top s ⇔
          s ⊆ topspace top ∧ open_in top (topspace top DIFF s)
   
   [closure_of]  Definition
      
      ⊢ ∀top s.
          top closure_of s =
          {x |
           x ∈ topspace top ∧ ∀t. x ∈ t ∧ open_in top t ⇒ ∃y. y ∈ s ∧ y ∈ t}
   
   [compact_in]  Definition
      
      ⊢ ∀top s.
          compact_in top s ⇔
          s ⊆ topspace top ∧
          ∀U. (∀u. u ∈ U ⇒ open_in top u) ∧ s ⊆ BIGUNION U ⇒
              ∃V. FINITE V ∧ V ⊆ U ∧ s ⊆ BIGUNION V
   
   [compact_space]  Definition
      
      ⊢ ∀top. compact_space top ⇔ compact_in top (topspace top)
   
   [continuous_map]  Definition
      
      ⊢ ∀top top' f.
          continuous_map (top,top') f ⇔
          (∀x. x ∈ topspace top ⇒ f x ∈ topspace top') ∧
          ∀u. open_in top' u ⇒ open_in top {x | x ∈ topspace top ∧ f x ∈ u}
   
   [derived_set_of]  Definition
      
      ⊢ ∀top s.
          top derived_set_of s =
          {x |
           x ∈ topspace top ∧
           ∀t. x ∈ t ∧ open_in top t ⇒ ∃y. y ≠ x ∧ y ∈ s ∧ y ∈ t}
   
   [frontier_of]  Definition
      
      ⊢ ∀top s. top frontier_of s = top closure_of s DIFF top interior_of s
   
   [hull]  Definition
      
      ⊢ ∀P s. P hull s = BIGINTER {t | P t ∧ s ⊆ t}
   
   [interior_of]  Definition
      
      ⊢ ∀top s. top interior_of s = {x | ∃t. open_in top t ∧ x ∈ t ∧ t ⊆ s}
   
   [istopology]  Definition
      
      ⊢ ∀L. istopology L ⇔
            ∅ ∈ L ∧ (∀s t. s ∈ L ∧ t ∈ L ⇒ s ∩ t ∈ L) ∧
            ∀k. k ⊆ L ⇒ BIGUNION k ∈ L
   
   [limpt]  Definition
      
      ⊢ ∀top x S'.
          limpt top x S' ⇔
          x ∈ topspace top ∧ ∀N. neigh top (N,x) ⇒ ∃y. x ≠ y ∧ S' y ∧ N y
   
   [neigh]  Definition
      
      ⊢ ∀top N x.
          neigh top (N,x) ⇔
          ∃P. open_in top P ∧ P ⊆ N ∧ P x ∧ N ⊆ topspace top
   
   [open_DEF]  Definition
      
      ⊢ ∀s. open s ⇔ open_in s 𝕌(:α)
   
   [pairwise]  Definition
      
      ⊢ ∀r s. pairwiseD r s ⇔ ∀x y. x ∈ s ∧ y ∈ s ∧ x ≠ y ⇒ r x y
   
   [relative_to]  Definition
      
      ⊢ ∀P s t. (P relative_to s) t ⇔ ∃u. P u ∧ s ∩ u = t
   
   [subtopology]  Definition
      
      ⊢ ∀top u. subtopology top u = topology {s ∩ u | open_in top s}
   
   [topcontinuous_at]  Definition
      
      ⊢ ∀top top' f x.
          topcontinuous_at top top' f x ⇔
          x ∈ topspace top ∧ (∀x. x ∈ topspace top ⇒ f x ∈ topspace top') ∧
          ∀v. open_in top' v ∧ f x ∈ v ⇒
              ∃u. open_in top u ∧ x ∈ u ∧ ∀y. y ∈ u ⇒ f y ∈ v
   
   [topology_TY_DEF]  Definition
      
      ⊢ ∃rep. TYPE_DEFINITION istopology rep
   
   [topology_tybij]  Definition
      
      ⊢ (∀a. topology (open_in a) = a) ∧
        ∀r. istopology r ⇔ open_in (topology r) = r
   
   [topspace]  Definition
      
      ⊢ ∀top. topspace top = BIGUNION {s | open_in top s}
   
   [ARBITRARY_INTERSECTION_OF_COMPLEMENT]  Theorem
