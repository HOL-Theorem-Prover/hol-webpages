   [Delta_def]  Definition

          SPO A B C f g =
          (partition (SPOr A f g) (B ⊔ C),
           restr (λb. equiv_class (SPOr A f g) (B ⊔ C) (INL b)) B,
           restr (λc. equiv_class (SPOr A f g) (B ⊔ C) (INR c)) C)
