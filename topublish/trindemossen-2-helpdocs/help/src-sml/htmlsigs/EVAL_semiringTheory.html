   [is_semi_ring_def]  Definition
      
      ⊢ ∀r. is_semi_ring r ⇔
            (∀n m. r.SRP n m = r.SRP m n) ∧
            (∀n m p. r.SRP n (r.SRP m p) = r.SRP (r.SRP n m) p) ∧
            (∀n m. r.SRM n m = r.SRM m n) ∧
            (∀n m p. r.SRM n (r.SRM m p) = r.SRM (r.SRM n m) p) ∧
            (∀n. r.SRP r.SR0 n = n) ∧ (∀n. r.SRM r.SR1 n = n) ∧
