   [HEX]  Definition
      
      ⊢ ∀n. HEX n =
            if n < 10 then CHR (ORD #"0" + n)
            else if n < 16 then CHR (ORD #"A" + (n − 10))
            else #"\^@"
   
   [UNHEX]  Definition
      
      ⊢ ∀c. UNHEX c =
            (let
               n = ORD c
             in
               if ORD #"0" ≤ n ∧ n ≤ ORD #"9" then n − ORD #"0"
               else if ORD #"a" ≤ n ∧ n ≤ ORD #"f" then 10 + n − ORD #"a"
               else if ORD #"A" ≤ n ∧ n ≤ ORD #"F" then 10 + n − ORD #"A"
               else 0)
   
   [fromBinString_def]  Definition
      
      ⊢ ∀s. fromBinString s =
            if s ≠ "" ∧ EVERY (λc. c = #"0" ∨ c = #"1") s then
              SOME (num_from_bin_string s)
            else NONE
   
   [fromDecString_def]  Definition
      
      ⊢ ∀s. fromDecString s =
            if s ≠ "" ∧ EVERY isDigit s then SOME (toNum s) else NONE
   
   [fromHexString_def]  Definition
      
      ⊢ ∀s. fromHexString s =
            if s ≠ "" ∧ EVERY isHexDigit s then
              SOME (num_from_hex_string s)
            else NONE
   
   [n2s_def]  Definition
      
      ⊢ ∀b f n. n2s b f n = REVERSE (MAP f (n2l b n))
