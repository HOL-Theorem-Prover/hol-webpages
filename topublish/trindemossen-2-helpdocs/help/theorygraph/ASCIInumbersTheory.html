HEX 
⊢ ∀n. HEX n =
      if n < 10 then CHR (ORD #"0" + n)
      else if n < 16 then CHR (ORD #"A" + (n − 10))
      else #"\^@"
UNHEX 
⊢ ∀c. UNHEX c =
      (let
         n = ORD c
       in
         if ORD #"0" ≤ n ∧ n ≤ ORD #"9" then n − ORD #"0"
         else if ORD #"a" ≤ n ∧ n ≤ ORD #"f" then 10 + n − ORD #"a"
         else if ORD #"A" ≤ n ∧ n ≤ ORD #"F" then 10 + n − ORD #"A"
         else 0)
