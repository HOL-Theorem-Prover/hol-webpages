FRAC_EQ 
|- ∀a1 b1 a2 b2.
     0 < b1 ⇒
     0 < b2 ⇒
     ((abs_frac (a1,b1) = abs_frac (a2,b2)) ⇔ (a1 = a2) ∧ (b1 = b2))
