|- SUM_REL_tupled =
   WFREC (@R. WF R)
     (\SUM_REL_tupled a.
        case a of
           (R1,R2,INL a1,INL a2) -> I (R1 a1 a2)
        || (R1,R2,INL a1,INR b2') -> I F
        || (R1,R2,INR b1,INL a2') -> I F
        || (R1,R2,INR b1,INR b2) -> I (R2 b1 b2))
