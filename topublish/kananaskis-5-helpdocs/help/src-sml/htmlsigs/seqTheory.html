   [nets] Parent theory of "seq"
   
   [cauchy]  Definition
      
      |- !f.
           cauchy f <=>
           !e. 0 < e ==> ?N. !m n. m >= N /\ n >= N ==> abs (f m - f n) < e
   
   [convergent]  Definition
      
      |- !f. convergent f <=> ?l. f --> l
   
   [lim]  Definition
      
      |- !f. lim f = @l. f --> l
   
   [mono]  Definition

           mono f <=>
           (!m n. m <= n ==> f m <= f n) \/ !m n. m <= n ==> f m >= f n

   [summable]  Definition

   [sums]  Definition
      
      |- !f s. f sums s <=> (\n. sum (0,n) f) --> s

           !a b. a <= b ==> P (a,b)

   [SEQ_ABS]  Theorem
      
      |- !f. (\n. abs (f n)) --> 0 <=> f --> 0
   
   [SEQ_ABS_IMP]  Theorem
      
      |- !f l. f --> l ==> (\n. abs (f n)) --> abs l
