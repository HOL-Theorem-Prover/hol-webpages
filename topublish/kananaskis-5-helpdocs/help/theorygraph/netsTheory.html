Definitions

dorder 
|- !g.
     dorder g <=>
     !x y. g x x /\ g y y ==> ?z. g z z /\ !w. g w z ==> g w x /\ g w y
tends 
|- !s l top g.
     (s tends l) (top,g) <=>
     !N. neigh top (N,l) ==> ?n. g n n /\ !m. g m n ==> N (s m)
bounded 
|- !m g f.
     bounded (m,g) f <=> ?k x N. g N N /\ !n. g n N ==> dist m (f n,x) < k
tendsto 
|- !m x y z.
     tendsto (m,x) y z <=> 0 < dist m (x,y) /\ dist m (x,y) <= dist m (x,z)
