iSQRTd_def 
|- !x n.
     iSQRTd (x,n) =
     (let p = SQRTd n in
      let next = 4 * SND p + x in
      let ndiff = 4 * FST p + 1 in
        if next < ndiff then
          (2 * FST p,next)
        else
          (2 * FST p + 1,next - ndiff))

|- !n.

iSQRT1_def 
|- !n.
     iSQRT1 n =
     (let p = SQRTd n in
      let d = SUC (SND p) - FST p in
        if d = 0 then
          (2 * FST p,SUC (4 * SND p))
        else
          (SUC (2 * FST p),4 * (d - 1)))
iSQRT2_def 
|- !n.
     iSQRT2 n =
     (let p = SQRTd n in
      let d = 2 * FST p in
      let c = SUC (2 * SND p) in
      let e = c - d in
        if e = 0 then (d,2 * c) else (SUC d,2 * e - 1))
iSQRT3_def 
|- !n.
     iSQRT3 n =
     (let p = SQRTd n in
      let d = 2 * FST p in
      let c = SUC (2 * SND p) in
      let e = SUC c - d in
        if e = 0 then (d,SUC (2 * c)) else (SUC d,2 * (e - 1)))
