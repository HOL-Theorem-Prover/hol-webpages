    (∀a b.
       interval [(a,b)] ≠ ∅ ⇒

    (∀f s.
       (λx. f (ms' x)) has_bounded_variation_on IMAGE ms s ⇔
       f has_bounded_variation_on s) ∧
    ∀f s. vector_variation (IMAGE ms s) (λx. f (ms' x)) = vector_variation s f
