is_insert_def 
⊢ ∀frame_l frame_r R k x al al'.
    is_insert frame_l frame_r R k x al al' ⇔
    irreflexive R ∧ transitive R ⇒
    SORTED R (MAP FST al) ⇒
    (ALOOKUP al' = ALOOKUP ((k,x)::al)) ∧
    (frame_l ⇒ al ≠ [] ∧ (FST (HD al') = FST (HD al))) ∧

    SORTED R (MAP FST al')
is_lookup_def 
⊢ ∀fl fr R al x r.

    ∀xs ys.
      fl ∨ (xs = []) ⇒
      fr ∨ (ys = []) ⇒
      irreflexive R ∧ transitive R ⇒
      SORTED R (MAP FST (xs ++ al ++ ys)) ⇒
      (ALOOKUP (xs ++ al ++ ys) x = r)
