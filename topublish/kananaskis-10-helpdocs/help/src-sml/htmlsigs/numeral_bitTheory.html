   [BIT_MODF_def]  Definition

      |- (∀f x b e y. BIT_MODF 0 f x b e y = y) ∧
         ∀n f x b e y.
           BIT_MODF (SUC n) f x b e y =
           BIT_MODF n f (x DIV 2) (b + 1) (2 * e)
             (if f b (ODD x) then e + y else y)

   [BIT_REV_def]  Definition

      |- (∀x y. BIT_REV 0 x y = y) ∧
