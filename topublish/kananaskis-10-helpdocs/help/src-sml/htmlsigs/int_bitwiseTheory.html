      |- bits_bitwise_tupled =
         WFREC

                R (f,(bs1,r1),bs2,r2) (f,(b1::bs1,r1),b2::bs2,r2))
           (λbits_bitwise_tupled a.
              case a of
                (f,([],r1),[],r2) => I ([],f r1 r2)
              | (f,(b1::bs1,r1),[],r2) =>

      |- ∀i.
           bits_of_int i =
           if i < 0 then (MAP $~ (bits_of_num (Num (int_not i))),T)
           else (bits_of_num (Num i),F)

   [int_and_def]  Definition

   [int_bit_def]  Definition
